#include<avr/io.h>
#define F_CPU 8000000UL
#include<util/delay.h>
unsigned char a[]={0xf9,0xa4,0xb0,0x99,0x92,0x82,0xf8,0x80,0x90};
int i;	

int main()
{
	DDRD&=!(1<<PD6);
	DDRD|=(1<<PD0);
    DDRB=0xff;
	DDRA=0xff;
	int i;
	while(1)
	{
		PORTB=0xfe;
		{
		for(i=0;i<10;i++)
		{
		PORTA=a[i];
		if(!(PIND&(1<<PD6)))
		{
		PORTD&=!(1<<PD0);
		}
		else
		{
		PORTD=(1<<PD0); 
		}
		_delay_ms(1000);
		}
		}
		PORTD|=(1<<PD0);
		
		PORTB=0xfd;
		for(i=0;i<2;i++)
		_delay_ms(1000);
		PORTB=0xfb;
		{
		for(i=0;i<10;i++)
		{
			PORTA=a[i];
			_delay_ms(1000);
		}
		}
	}
return 0;
}
